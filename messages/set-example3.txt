Example of an Invalid Parameter with allow_partial set to False

::Request::

header {
  msg_id: "8880"
  msg_type: SET
  proto_version: "1.0"
  to_id: "usp.00D09E-RPi_Test-T0000000001"
  from_id: "usp.controller-stomp-johnb"
  reply_to_id: "usp.controller-stomp-johnb"
}
body {
  request {
    set {
      update_obj {
        obj_path: "Device.LocalAgent."
        param_setting {
          param: "PeriodicIntervalZZZ"
          value: "60"
          required: true
        }
      }
    }
  }
}



::Response::

header {
  msg_id: "8880"
  proto_version: "1.0"
  to_id: "usp.controller-stomp-johnb"
  from_id: "usp.00D09E-RPi_Test-T0000000001"
}
body {
  error {
    err_code: 9000
    err_msg: "Invalid Path Found, Allow Partial Updates = False :: Fail the entire Set"
    param_err {
      param_path: "Device.LocalAgent."
      err_code: 9000
      err_msg: "Failed to Set Required Parameters: Device.LocalAgent.PeriodicIntervalZZZ "
    }
  }
}
